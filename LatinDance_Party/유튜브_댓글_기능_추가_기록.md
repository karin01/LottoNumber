#### 댓글 로드 함수
```javascript
// 유튜브 댓글 로드
async function loadYouTubeComments(videoId) {
    const commentsContainer = document.getElementById('youtube-comments-container');
    
    try {
        let comments = [];
        
        // Firebase가 초기화되었는지 확인
        if (db) {
            // Firebase에서 댓글 로드
            const commentsSnapshot = await db.collection('youtubeComments')
                .where('videoId', '==', videoId)
                .orderBy('createdAt', 'desc')
                .get();
            
            comments = commentsSnapshot.docs.map(doc => ({
                id: doc.id,
                ...doc.data()
            }));
        } else {
            // 로컬 스토리지에서 댓글 로드
            const allComments = JSON.parse(localStorage.getItem('youtubeComments') || '[]');
            comments = allComments.filter(comment => comment.videoId === videoId);
        }
        
        displayYouTubeComments(comments);
        
    } catch (error) {
        console.error('댓글 로드 실패:', error);
        showMessage('댓글을 불러오는데 실패했습니다.', 'error');
    }
}
```

#### 댓글 작성 함수
```javascript
// 유튜브 댓글 작성
async function submitYouTubeComment() {
    // 로그인 확인
    if (!currentUser) {
        showMessage('로그인이 필요한 기능입니다.', 'error');
        return;
    }
    
    const commentText = document.getElementById('youtube-comment-text').value.trim();
    if (!commentText) {
        showMessage('댓글 내용을 입력해주세요.', 'error');
        return;
    }
    
    const newComment = {
        videoId: video.id,
        text: commentText,
        author: currentUser.displayName || currentUser.email,
        authorId: currentUser.uid,
        createdAt: new Date().toISOString()
    };
    
    try {
        // Firebase 또는 로컬 스토리지에 저장
        if (db) {
            const docRef = await db.collection('youtubeComments').add(newComment);
            newComment.id = docRef.id;
        } else {
            const allComments = JSON.parse(localStorage.getItem('youtubeComments') || '[]');
            newComment.id = Date.now().toString();
            allComments.push(newComment);
            localStorage.setItem('youtubeComments', JSON.stringify(allComments));
        }
        
        // 댓글 입력창 초기화 및 목록 새로고침
        document.getElementById('youtube-comment-text').value = '';
        await loadYouTubeComments(video.id);
        
        showMessage('댓글이 성공적으로 작성되었습니다!', 'success');
        
    } catch (error) {
        console.error('댓글 작성 실패:', error);
        showMessage('댓글 작성에 실패했습니다: ' + error.message, 'error');
    }
}
```

    if (!confirm('정말로 이 댓글을 삭제하시겠습니까?')) {
        return;
    }
    
    try {
        // Firebase 또는 로컬 스토리지에서 삭제
        if (db) {
            await db.collection('youtubeComments').doc(commentId).delete();
        } else {
            const allComments = JSON.parse(localStorage.getItem('youtubeComments') || '[]');
            const updatedComments = allComments.filter(comment => comment.id !== commentId);

## 기능 특징

### 1. 사용자 인증 기반
- 로그인한 사용자만 댓글 작성 가능
- 댓글 작성자와 관리자만 댓글 삭제 가능

### 2. 실시간 데이터 동기화
- Firebase Firestore 연동 (프로덕션)
- 로컬 스토리지 백업 (개발 환경)

### 3. 사용자 친화적 UI
- 직관적인 댓글 작성 폼
- 댓글 작성자와 날짜 표시
- 삭제 버튼 (권한 있는 사용자만)

### 4. 모바일 최적화
- 반응형 디자인
- iOS 줌 방지 (font-size: 16px)
- 터치 친화적 버튼 크기
