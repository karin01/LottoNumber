# 전화번호 인증 구현 가이드

## 📋 필요 단계

### 1. Firebase Phone Auth 설정

#### A. Firebase Console에서 설정
1. **Firebase Console 접속**
   - https://console.firebase.google.com
   - 프로젝트 선택

2. **Authentication → Sign-in method**
   - "전화번호" 제공업체 활성화
   - "테스트 전화번호" 추가 (개발용)

3. **reCAPTCHA 설정**
   - "reCAPTCHA Enterprise" 또는 "reCAPTCHA v3" 설정
   - 도메인 등록: `latinmat.co.kr`, `localhost`

### 2. 코드 수정

#### A. HTML에 reCAPTCHA 추가
```html
<!-- index.html의 <head> 섹션에 추가 -->
```

#### B. JavaScript 코드 수정
```javascript
// main.js의 signInWithPhone 함수 수정
let confirmationResult = null;

async function signInWithPhone(event) {
    event.preventDefault();
    
    if (!auth) {
        console.error('Firebase Auth가 초기화되지 않았습니다.');
        return;
    }
    
    const phoneNumber = document.getElementById('login-phone').value;
    
    // 전화번호 형식 검증
    const phoneRegex = /^01[0-9]-\d{3,4}-\d{4}$/;
    if (!phoneRegex.test(phoneNumber)) {
        showToast('올바른 전화번호 형식을 입력해주세요. (예: 010-1234-5678)', 'error');
        return;
    }
    
    try {
        // reCAPTCHA 실행
        const recaptchaToken = await grecaptcha.execute('YOUR_RECAPTCHA_SITE_KEY', {action: 'phone_auth'});
        
        // reCAPTCHA 검증
        const appVerifier = new firebase.auth.RecaptchaVerifier('phone-login-form', {
            'size': 'invisible',
            'callback': (response) => {
                console.log('reCAPTCHA 성공:', response);
            },
            'expired-callback': () => {
                showToast('reCAPTCHA가 만료되었습니다. 다시 시도해주세요.', 'error');
            }
        }, auth);
        
        // SMS 전송
        confirmationResult = await auth.signInWithPhoneNumber(phoneNumber, appVerifier);
        
        // 인증번호 입력 모달 표시
        showVerificationModal();
        
        showToast('인증번호가 전송되었습니다.', 'success');
    } catch (error) {
        console.error('전화번호 인증 실패:', error);
        showToast('인증번호 전송에 실패했습니다: ' + error.message, 'error');
    }
}

// 인증번호 확인 모달 표시
function showVerificationModal() {
    const modal = document.createElement('div');
    modal.className = 'modal';
    modal.id = 'verification-modal';
    modal.innerHTML = `
        <div class="modal-content verification-modal">
            <div class="modal-header">
                <h3>📱 인증번호 확인</h3>
                <button class="close-btn" onclick="closeVerificationModal()">&times;</button>
            </div>
            <div class="modal-body">
                <p>전송된 SMS 인증번호를 입력해주세요.</p>
                <form id="verification-form" onsubmit="verifyCode(event)">
                    <input type="text" id="verification-code" placeholder="인증번호 6자리" maxlength="6" required>
                    <button type="submit" class="verify-btn">확인</button>
                </form>
                <div class="verification-actions">
                    <button class="resend-btn" onclick="resendCode()">인증번호 재전송</button>
                    <button class="change-phone-btn" onclick="changePhoneNumber()">전화번호 변경</button>
                </div>
            </div>
        </div>
    `;
    document.body.appendChild(modal);
}

// 인증번호 확인
async function verifyCode(event) {
    event.preventDefault();
    
    const code = document.getElementById('verification-code').value;
    
    if (!confirmationResult) {
        showToast('인증 세션이 만료되었습니다. 다시 시도해주세요.', 'error');
        return;
    }
    
    try {
        const result = await confirmationResult.confirm(code);
        console.log('전화번호 인증 성공:', result.user);
        
        closeVerificationModal();
        closeLoginModal();
        showToast('전화번호로 로그인되었습니다!', 'success');
    } catch (error) {
        console.error('인증번호 확인 실패:', error);
        showToast('인증번호가 올바르지 않습니다.', 'error');
    }
}

// 인증번호 재전송
async function resendCode() {
    if (!confirmationResult) {
        showToast('인증 세션이 만료되었습니다. 다시 시도해주세요.', 'error');
        return;
    }
    
    try {
        await confirmationResult.confirm('000000'); // 임시 코드로 재전송 트리거
        showToast('인증번호가 재전송되었습니다.', 'success');
    } catch (error) {
        console.error('인증번호 재전송 실패:', error);
        showToast('인증번호 재전송에 실패했습니다.', 'error');
    }
}

// 전화번호 변경
function changePhoneNumber() {
    closeVerificationModal();
    // 로그인 모달로 돌아가기
}

// 인증 모달 닫기
function closeVerificationModal() {
    const modal = document.getElementById('verification-modal');
    if (modal) {
        modal.remove();
    }
}
```

### 4. 환경 변수 설정

#### A. .env 파일에 추가
```bash
# .env 파일
RECAPTCHA_SITE_KEY=your_recaptcha_site_key_here
RECAPTCHA_SECRET_KEY=your_recaptcha_secret_key_here
```

#### B. Firebase 설정
```bash
firebase functions:config:set recaptcha.site_key="your_site_key"
firebase functions:config:set recaptcha.secret_key="your_secret_key"
```
